# install package

```bash
sudo apt install build-essential pkg-config \
  libgl1-mesa-dev libx11-dev libxcursor-dev libxrandr-dev \
  libxinerama-dev libxi-dev libxxf86vm-dev libasound2-dev
```

or

```bash
sudo apt update
sudo apt install -y xorg-dev
```
